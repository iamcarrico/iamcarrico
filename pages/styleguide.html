Syntax Highlighting — JavaScript
Prism.js at build time — colors update with dark mode
{% highlight "js" %} /** * Debounce — delays invoking fn until after `wait` ms have elapsed * since the last time it was called. */ function debounce(fn, wait = 300) { let timer; return function (...args) { clearTimeout(timer); timer = setTimeout(() => fn.apply(this, args), wait); }; } // Example: debounce a scroll listener so it only fires 200ms after scrolling stops const SCROLL_THRESHOLD = 120; const onScroll = debounce(function handleScroll(event) { const scrollY = window.scrollY || document.documentElement.scrollTop; const header = document.querySelector('.site-header'); const isCondensed = header.classList.contains('condensed'); if (scrollY > SCROLL_THRESHOLD && !isCondensed) { header.classList.add('condensed'); } else if (scrollY <= SCROLL_THRESHOLD && isCondensed) { header.classList.remove('condensed'); } }, 200); window.addEventListener('scroll', onScroll, { passive: true }); {% endhighlight %}
